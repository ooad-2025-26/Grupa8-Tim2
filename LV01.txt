#include <iostream> 
 
 unsigned long long int Faktorijel(int n) { 
   if(n < 0) throw n; 
  unsigned long long int p = 1; 
   for(int i = 1; i <= n; i++) p *= i; 
   return p; 
 }  
 
 unsigned long long int BinomniKoeficijent(int n, int k) { 
   try { 
     return Faktorijel(n) / (Faktorijel(k) * Faktorijel(n - k)); 
   } 
   catch(int e) { 
     return 0;     // Regularan izlazak iz funkcije... 
   }  
 }  
 
 int main() { 
   int n, k; 
   std::cin >> n >> k;  
   std::cout << BinomniKoeficijent(n, k) << std::endl;  
  return 0; 
 }
